** Thresholds
Set gc-cons-threshold Smaller for Interactive Use

A large gc-cons-threshold may cause freezing and stuttering during long-term interactive use.

If you experience freezing, decrease this amount, if you experience stuttering, increase this amount.
#+begin_src emacs-lisp

  If you experience freezing, decrease this.  If you experience stuttering, increase this.")

              (setq gc-cons-threshold better-gc-cons-threshold)

** Daily Wellbeing Check-in
Automated morning check-in that pops up on startup and during idle
time.  Entries are stored in =~/orgfiles/daily-log.org= as headed
org entries with machine-readable properties, making them easy to
analyse with org-table spreadsheets, Python/pandas, etc.

Tracked fields:
- *Last night*: sleep hours, sleep quality
- *Right now*: energy, mood, stress, anxiety
- *Yesterday*: exercise, food quality, breakfast, lunch, dinner,
  tea, coffee, alcohol, deep-work hours, productivity, social quality
- *Reflections*: best thing, main challenge, gratitude,
  what is lacking/better, what is getting worse, free notes

Key bindings inside the check-in buffer:
| =C-c C-c= | Save & close                 |
| =C-c s=   | Snooze (prompts for minutes) |
| =C-c C-k= | Cancel (entry not saved)     |
| =C-c m d= | Open manually from anywhere  |

#+begin_src emacs-lisp
  ;;; ──────────────────────────────────────────────────────────────────
  ;;; Daily Wellbeing & Metrics Check-in
  ;;; ──────────────────────────────────────────────────────────────────

  (defvar mz/daily-log-file "~/orgfiles/daily-log.org"
    "Org file where daily wellbeing entries are stored.")

  (defvar mz/daily-log-idle-timer nil
    "Repeating idle timer handle for the morning check-in popup.")

  (defvar mz/daily-log-snooze-timer nil
    "One-shot timer handle when a snooze is active.")

  (defconst mz/daily-log-morning-start 6
    "First hour (0-23) of the morning popup window.")

  (defconst mz/daily-log-morning-end 11
    "Last hour (exclusive, 0-23) of the morning popup window.")

  (defconst mz/daily-log-idle-seconds 300
    "Seconds of idle time before the morning check-in popup is shown.")

  ;;; ── Predicates ────────────────────────────────────────────────────

  (defun mz/daily-log-today ()
    "Return today's date as YYYY-MM-DD."
    (format-time-string "%Y-%m-%d"))

  (defun mz/daily-log-logged-p ()
    "Return non-nil if a check-in entry for today already exists."
    (let ((f (expand-file-name mz/daily-log-file)))
      (when (file-exists-p f)
        (with-temp-buffer
          (insert-file-contents f)
          (goto-char (point-min))
          (re-search-forward
           (concat ":DAILY_DATE:[[:space:]]+" (regexp-quote (mz/daily-log-today)))
           nil t)))))

  (defun mz/daily-log-normalize-key (raw)
    "Normalize RAW key text from form labels to canonical storage keys."
    (let* ((k (downcase (string-trim raw)))
           (k (replace-regexp-in-string "[^[:alnum:]]+" "_" k))
           (k (replace-regexp-in-string "_+" "_" k))
           (k (replace-regexp-in-string "^_\\|_$" "" k)))
      (pcase k
        ((or "what_am_i_lacking_and_what_could_be_better"
             "what_is_lacking_and_what_could_be_better"
             "lacking_and_better")
         "lacking_and_better")
        ((or "what_is_getting_worse" "getting_worse") "getting_worse")
        ((or "deep_work_hrs" "deep_work_hour" "deep_work") "deep_work_hours")
        (_ k))))

  (defun mz/daily-log-morning-window-p ()
    "Return non-nil if the current hour is within the morning window."
    (let ((hour (nth 2 (decode-time))))
      (and (>= hour mz/daily-log-morning-start)
           (<  hour mz/daily-log-morning-end))))

    (concat
     "# ════════════════════════════════════════════════════════════\n"
     "# Daily Wellbeing Check-in  ·  " (mz/daily-log-today) "\n"
     "#\n"
     "# Fill each field after the colon.  Scales: 1 (worst) – 10 (best).\n"
     "# C-c C-c  Save & close     C-c s  Snooze     C-c C-k  Cancel\n"
     "# ════════════════════════════════════════════════════════════\n"
     "\n"
     "──── LAST NIGHT ─────────────────────────────────────────────\n"
     "sleep_hours:        \n"
     "sleep_quality:      \n"
     "\n"
     "──── RIGHT NOW ──────────────────────────────────────────────\n"
     "energy:             \n"
     "mood:               \n"
     "stress:             \n"
     "anxiety:            \n"
     "\n"
     "──── YESTERDAY ──────────────────────────────────────────────\n"
     "exercise:           \n"
     "food_quality:       \n"
    "breakfast:          \n"
    "lunch:              \n"
    "dinner:             \n"
    "tea:                \n"

  (defun mz/daily-log--find-entry-block (&optional date)
    "Return latest check-in entry block as string.
If DATE is non-nil, return latest block matching DATE (YYYY-MM-DD)."
    (let ((f (expand-file-name mz/daily-log-file))
          (found nil))
      (when (file-exists-p f)
        (with-temp-buffer
          (insert-file-contents f)
          (goto-char (point-min))
          (while (re-search-forward "^\\* Check-in " nil t)
            (let* ((beg (line-beginning-position))
                   (end (or (save-excursion
                              (when (re-search-forward "^\\* Check-in " nil t)
                                (line-beginning-position)))
                            (point-max)))
                   (block (buffer-substring-no-properties beg end)))
              (when (or (null date)
                        (string-match-p
                         (format ":DAILY_DATE:[[:space:]]*%s" (regexp-quote date))
                         block))
                (setq found block))
              (goto-char end)))))
      found))

    "Extract first capture group from BLOCK with REGEXP; return empty string on miss."
    (if (and block (string-match regexp block))
        (string-trim (match-string 1 block))
      ""))

  (defun mz/daily-log-parse ()
    "Parse the check-in buffer; return list of (KEY . VALUE) cons cells."
    (let (result)
      (save-excursion
        (goto-char (point-min))
        (while (re-search-forward
          "^\\([^#\n:][^:\n]*\\):[ \t]*\\(.*\\)$" nil t)
        (let ((k (mz/daily-log-normalize-key (match-string-no-properties 1)))
                (v (string-trim (match-string-no-properties 2))))
            (push (cons k v) result))))
      (nreverse result)))

  (defun mz/daily-log-submit ()
    "Parse, save, and close the check-in buffer."
    (interactive)
    (let ((pairs (mz/daily-log-parse)))
      (cond
       ((null pairs)
        (message "No fields found – the template may be malformed."))
       ((cl-every (lambda (p) (string-empty-p (cdr p))) pairs)
        (when (yes-or-no-p "All fields are empty – save anyway? ")
          (mz/daily-log-save pairs)
          (quit-window t)))
       (t
        (mz/daily-log-save pairs)
        (quit-window t)
        (message "Daily check-in saved for %s." (mz/daily-log-today))))))

  (defun mz/daily-log-cancel ()
    "Close the check-in buffer without saving."
    (interactive)
    (quit-window t)
    (message "Check-in cancelled – will reappear next morning."))

  (defun mz/daily-log-snooze ()
    "Snooze the check-in for a user-specified number of minutes."
    (interactive)
    (let* ((raw  (read-string "Snooze for how many minutes? [30]: " nil nil "30"))
           (mins (string-to-number raw)))
      (if (<= mins 0)
          (message "Invalid snooze duration.")
        (when (timerp mz/daily-log-snooze-timer)
          (cancel-timer mz/daily-log-snooze-timer))
        (setq mz/daily-log-snooze-timer
              (run-at-time (* mins 60) nil #'mz/daily-log-open))
        (quit-window t)
        (message "Check-in snoozed for %d minute(s)." mins))))

  (defun mz/daily-log-open (&optional force)
    "Open the daily check-in buffer.
When FORCE is non-nil, open even if today is already logged."
    (interactive)
    (if (and (not force) (mz/daily-log-logged-p))
        (when (called-interactively-p 'any)
          (message "Already logged today (%s)." (mz/daily-log-today)))
      ;; Reuse any existing buffer to avoid wiping in-progress entry
      (let ((existing (get-buffer "*Daily Check-in*")))
        (if existing
            (pop-to-buffer existing)
          (let ((buf (get-buffer-create "*Daily Check-in*")))
            (with-current-buffer buf
              (let ((inhibit-read-only t))
                (erase-buffer)
                (insert (mz/daily-log-template-with-defaults))
                (goto-char (point-min))
                (re-search-forward "sleep_hours:[[:space:]]+" nil t))
              (mz/daily-log-mode 1))
            (pop-to-buffer buf))))))

  (defun mz/daily-log-maybe-open ()
    "Open check-in only during the morning window and when not yet logged."
    (when (and (mz/daily-log-morning-window-p)
               (not (mz/daily-log-logged-p)))
      (mz/daily-log-open)))

  (defun mz/daily-log-setup ()
    "Run the check-in immediately at startup; install a repeating idle timer."

    (setq mz/daily-log-idle-timer

    "Run consult-line with the current function name in opx2-js-mode.

    (interactive)
    (when (and (derived-mode-p 'opx2-js-mode)
           (fboundp 'which-function))

             (match-string 1 function-name)

  (defun mz/export-org-to-txt-home ()
    (when-let* ((visiting-file (buffer-file-name))
                (file-dir (file-name-directory visiting-file))
                (_ (and (equal major-mode #'org-mode)
                        (some (lambda (x)
                                (string-equal (expand-file-name x) file-dir))
